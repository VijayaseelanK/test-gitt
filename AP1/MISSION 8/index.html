1. Convertir Celsius en Fahrenheit 
function convertirTemperature() {
    const celsius = 25;
    const fahrenheit = (celsius * 9 / 5) + 32;
    alert(`${celsius}°C = ${fahrenheit}°F`);
}
2. Ajuster la taille du texte

Ceci est un texte ajustable.

function augmenterTaille() {
    let texte = document.getElementById('texte');
    let taille = parseInt(window.getComputedStyle(texte).fontSize);
    texte.style.fontSize = (taille + 2) + 'px';
}

function reduireTaille() {
    let texte = document.getElementById('texte');
    let taille = parseInt(window.getComputedStyle(texte).fontSize);
    texte.style.fontSize = (taille - 2) + 'px';
}

4. Afficher une salutation selon l'heure 
function saluerSelonHeure() {
    const heure = new Date().getHours();
    let salutation = '';
    if (heure < 12) {
        salutation = 'Bonjour';
    } else if (heure < 18) {
        salutation = 'Bon après-midi';
    } else {
        salutation = 'Bonsoir';
    }
    alert(salutation);
}
5. Afficher l'heure actuelle 
function afficherHeure() {

}
6. Calculer l'âge à partir de la date de naissance 
function calculerAge() {
    const naissance = prompt("Entrez votre année de naissance (ex: 1990):");
    const anneeActuelle = new Date().getFullYear();
    const age = anneeActuelle - naissance;
    alert(`Votre âge est : ${age} ans.`);
}
7. Simuler un réveil matin 
function simulerReveil() {
    const heure = parseInt(prompt("Entrez l'heure de réveil (ex: 7):"));
    const minutes = parseInt(prompt("Entrez les minutes (ex: 30):"));
    alert(`Réveil réglé à ${heure}h${minutes < 10 ? '0' : ''}${minutes}. Bonne nuit!`);
}
8. Calculer le pourboire 
function calculerPourboire() {
    const montant = parseFloat(prompt("Entrez le montant de l'addition (€):"));
    const pourboire = (montant * 0.15).toFixed(2);
    alert(`Le pourboire recommandé est : ${pourboire} €`);
}
9. Calculer les calories brûlées 
function calculerCalories() {
    const sport = prompt("Entrez le type d'activité (ex: marche, course):");
    const minutes = parseInt(prompt("Durée de l'activité (en minutes):"));
    const calories = sport === "marche" ? (minutes * 4) : (sport === "course" ? (minutes * 10) : 0);
    alert(`Vous avez brûlé environ ${calories} calories.`);
}
10. Vérifier si un nombre est pair ou impair 
function verifierPairImpair() {
    const nombre = parseInt(prompt("Entrez un nombre:"));
    const resultat = nombre % 2 === 0 ? "pair" : "impair";
    alert(`Le nombre ${nombre} est ${resultat}.`);
}
