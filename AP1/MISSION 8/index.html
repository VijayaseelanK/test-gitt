1. Convertir Celsius en Fahrenheit 
function convertirTemperature() {
    const celsius = 25;
    const fahrenheit = (celsius * 9 / 5) + 32;
    alert(`${celsius}°C = ${fahrenheit}°F`);
}
2. Ajuster la taille du texte

Ceci est un texte ajustable.

function augmenterTaille() {
    let texte = document.getElementById('texte');
    let taille = parseInt(window.getComputedStyle(texte).fontSize);
    texte.style.fontSize = (taille + 2) + 'px';
}

function reduireTaille() {
    let texte = document.getElementById('texte');
    let taille = parseInt(window.getComputedStyle(texte).fontSize);
    texte.style.fontSize = (taille - 2) + 'px';
}

4. Afficher une salutation selon l'heure 
function saluerSelonHeure() {
    const heure = new Date().getHours();
    let salutation = '';
    if (heure < 12) {
        salutation = 'Bonjour';
    } else if (heure < 18) {
        salutation = 'Bon après-midi';
    } else {
        salutation = 'Bonsoir';
    }
    alert(salutation);
}

6. Calculer l'âge à partir de la date de naissance 
function calculerAge() {
    const naissance = prompt("Entrez votre année de naissance (ex: 1990):");
    const anneeActuelle = new Date().getFullYear();
    const age = anneeActuelle - naissance;
    alert(`Votre âge est : ${age} ans.`);
}

11. Calculer la moyenne de plusieurs nombres 
function calculerMoyenne() {
    const nombres = prompt("Entrez plusieurs nombres séparés par des virgules:").split(',');
    const somme = nombres.reduce((acc, val) => acc + parseFloat(val), 0);
    const moyenne = somme / nombres.length;
    alert(`La moyenne des nombres est : ${moyenne}`);
}
12. Vérifier si un mot est un palindrome 
function verifierPalindrome() {
    const mot = prompt("Entrez un mot:").toLowerCase();
    const motInverse = mot.split('').reverse().join('');
    if (mot === motInverse) {
        alert(`${mot} est un palindrome.`);
    } else {
        alert(`${mot} n'est pas un palindrome.`);
    }
}
13. Calculer la factorielle d'un nombre 
function calculerFactorielle() {
    const nombre = parseInt(prompt("Entrez un nombre:"));
    let factorielle = 1;
    for (let i = 1; i <= nombre; i++) {
        factorielle *= i;
    }
    alert(`La factorielle de ${nombre} est : ${factorielle}`);
}
14. Trouver le plus grand nombre dans un tableau 
function trouverMax() {
    const nombres = [1, 5, 10, 3, 7];
    const max = Math.max(...nombres);
    alert(`Le plus grand nombre est : ${max}`);
}
15. Vérifier si un nombre est premier 
function verifierPremier() {
    const nombre = parseInt(prompt("Entrez un nombre:"));
    let estPremier = true;
    for (let i = 2; i < nombre; i++) {
        if (nombre % i === 0) {
            estPremier = false;
            break;
        }
    }
    if (estPremier && nombre > 1) {
        alert(`${nombre} est un nombre premier.`);
    } else {
        alert(`${nombre} n'est pas un nombre premier.`);
    }
}
