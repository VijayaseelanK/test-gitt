2. Ajuster la taille du texte

Ceci est un texte ajustable.

function augmenterTaille() {
    let texte = document.getElementById('texte');
    let taille = parseInt(window.getComputedStyle(texte).fontSize);
    texte.style.fontSize = (taille + 2) + 'px';
}

function reduireTaille() {
    let texte = document.getElementById('texte');
    let taille = parseInt(window.getComputedStyle(texte).fontSize);
    texte.style.fontSize = (taille - 2) + 'px';
}

4. Afficher une salutation selon l'heure 
function saluerSelonHeure() {
    const heure = new Date().getHours();
    let salutation = '';
    if (heure < 12) {
        salutation = 'Bonjour';
    } else if (heure < 18) {
        salutation = 'Bon après-midi';
    } else {
        salutation = 'Bonsoir';
    }
    alert(salutation);
}

10. Calculer la factorielle d'un nombre 
function calculerFactorielle() {
    let n = parseInt(document.getElementById('factorielle').value);
    let result = 1;
    for (let i = 1; i <= n; i++) {
        result *= i;
    }
    alert(\`La factorielle de \${n} est \${result}\`);
}

13. Trier une liste de nombres 
function trierNombres() {
    let nombres = document.getElementById('listeNombres').value.split(',').map(Number);
    nombres.sort((a, b) => a - b);
    alert(\`Les nombres triés sont : \${nombres.join(', ')}\`);
}
14. Créer un tableau de carrés de nombres 
function creerCarres() {
    let n = parseInt(document.getElementById('maxCarres').value);
    let carres = [];
    for (let i = 1; i <= n; i++) {
        carres.push(i * i);
    }
    alert(\`Les carrés des nombres sont : \${carres.join(', ')}\`);
}
15. Vérifier si un mot est un palindrome 
function verifierPalindrome() {
    let mot = document.getElementById('motPalindrome').value;
    let inverse = mot.split('').reverse().join('');
    if (mot === inverse) {
        alert('C\'est un palindrome');
    } else {
        alert('Ce n\'est pas un palindrome');
    }
}
